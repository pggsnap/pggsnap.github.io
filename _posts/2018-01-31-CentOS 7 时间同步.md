# 时间同步

```
# 安装ntp服务
yum install ntp
# 开机自启动
systemctl enable ntpd
# 启动服务
systemctl start ntpd
# 设置时区
timedatectl set-timezone Asia/Shanghai
# 启用ntp同步
timedatectl set-ntp yes
# 同步时间
ntpq -p
# 如需更改时间服务器，修改/etc/ntp.conf文件中的服务器地址即可.
```

```
[root@eureka2 ~]# vim /etc/ntp.conf
#server 0.centos.pool.ntp.org iburst
#server 1.centos.pool.ntp.org iburst
#server 2.centos.pool.ntp.org iburst
#server 3.centos.pool.ntp.org iburst
server 172.16.100.117 iburst
[root@eureka2 ~]# timedatectl set-ntp yes
[root@eureka2 ~]# systemctl start ntpd
[root@eureka2 ~]# systemctl enable ntpd
[root@eureka2 ~]# systemctl status ntpd
[root@eureka2 ~]# ntpq -p
     remote           refid      st t when poll reach   delay   offset  jitter
==============================================================================
 172.16.100.117  .LOCL.           1 u   53   64    7    0.324  1252.68   0.949
 [root@eureka2 ~]# timedatectl
```
